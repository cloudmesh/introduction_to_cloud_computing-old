[docs]    def image_list(self):
        images = {}
        lines = vagrant("box", "list")
        for line in lines:
            (name, kind) = line.split("(")
            name = name.strip()
            kind = kind.split(")")[0].strip()
            images[name] = {"name": name, "kind": kind}
        return images
