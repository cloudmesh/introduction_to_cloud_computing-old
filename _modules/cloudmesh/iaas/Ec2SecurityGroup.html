        def __eq__(self, other):

            if isinstance(other, Ec2SecurityGroup.Rule) and\

               self.to_port == other.to_port:
                ret = True
            return ret

        # '!=' comparison
        def __ne__(self, other):
            return not self.__eq__(other)
