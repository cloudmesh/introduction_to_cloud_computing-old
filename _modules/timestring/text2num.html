[docs]def text2num(s):
    a = re.split(r"[\s-]+", s.strip())
    n = 0
    g = 0
    for w in a:
        try:
            x = int(w)
        except:
            x = Small.get(w, None)
        finally:
            if x is not None:
                g += x
            elif w == "hundred":
                g *= 100
            else:
                x = Magnitude.get(w, None)
                if x is not None:
                    n += g * x
                    g = 0
                else:
                    raise NumberException("Unknown number: "+w)
    return n + g
